```typescript
@Injectable()
export class DocParserService {
  // DOCX 파싱 — mammoth 사용
  async parseDocx(buffer: Buffer): Promise<ParsedDocument> {
    const result = await mammoth.convertToHtml({ buffer });
    // HTML → 구조화된 섹션 추출
    return { type: 'docx', sections: this.extractSections(result.value), rawText: ... };
  }

  // PPTX 파싱 — jszip + XML 파싱
  async parsePptx(buffer: Buffer): Promise<ParsedDocument> {
    const zip = await JSZip.loadAsync(buffer);
    // ppt/slides/slide*.xml 순회하여 텍스트 추출
    return { type: 'pptx', slides: [...], rawText: ... };
  }

  // PDF 파싱 — pdf-parse 사용
  async parsePdf(buffer: Buffer): Promise<ParsedDocument> {
    const data = await pdfParse(buffer);
    return { type: 'pdf', pages: data.numpages, rawText: data.text };
  }

  // MIME 타입별 자동 라우팅
  async parse(buffer: Buffer, mimeType: string): Promise<ParsedDocument> {
    switch (mimeType) {
      case 'application/vnd.openxmlformats-officedocument.wordprocessingml.document':
        return this.parseDocx(buffer);
      case 'application/vnd.openxmlformats-officedocument.presentationml.presentation':
        return this.parsePptx(buffer);
      case 'application/pdf':
        return this.parsePdf(buffer);
      default:
        throw new BusinessException(ERROR_CODES.UNSUPPORTED_FILE_TYPE);
    }
  }
}
```

```typescript
export class DocExtractionService {
  constructor(
    private claudeService: ClaudeService,
    private docParser: DocParserService,
    private baseData: BaseDataService,
    private categoryRepo: Repository<DocBaseCategoryEntity>,
  ) {}

  // 단일 문서 추출 (Drive fileId로 직접)
  async extractFromDriveFile(entityId: string, fileId: string, userId: string): Promise<ExtractionResult> {
    // 1. Drive에서 다운로드
    const { stream, metadata } = await this.driveService.downloadFile(fileId);
    const buffer = await this.streamToBuffer(stream);

    // 2. 파싱
    const parsed = await this.docParser.parse(buffer, metadata.mimeType);

    // 3. 카테고리 스키마 로드
    const categories = await this.categoryRepo.find({ where: { entId: entityId } });

    // 4. Claude AI로 구조화 추출
    const extracted = await this.aiExtract(parsed, categories);

    return { sourceFile: metadata.name, categories: extracted, confidence: ... };
  }

  // 일괄 추출 (여러 문서)
  async batchExtract(entityId: string, fileIds: string[], userId: string): Observable<ExtractionProgress> {
    // SSE 스트리밍으로 진행률 반환
  }

  // AI 추출 핵심 로직
  private async aiExtract(parsed: ParsedDocument, categories: DocBaseCategoryEntity[]): Promise<ExtractedData[]> {
    const schemaPrompt = categories.map(c =>
      `카테고리: ${c.dbcCode}\n필드: ${JSON.stringify(c.dbcFieldSchema)}`
    ).join('\n\n');

    const response = await this.claudeService.sendMessage(
      EXTRACTION_SYSTEM_PROMPT,  // "기초데이터 카테고리 스키마에 맞게 문서에서 데이터를 추출하세요..."
      [{ role: 'user', content: `## 카테고리 스키마\n${schemaPrompt}\n\n## 문서 내용\n${parsed.rawText}` }]
    );

    return JSON.parse(response);
  }
}
```

#### A-2-3: GapAnalysisService (신규)

**파일**: `apps/api/src/domain/kms/service/doc-builder/gap-analysis.service.ts`

```typescript
@Injectable()
export class GapAnalysisService {
  // 카테고리별 필수 필드 대비 누락 분석
  async analyze(entityId: string, extracted: ExtractedData[]): Promise<GapReport> {
    const categories = await this.categoryRepo.find({ where: { entId: entityId } });
    const gaps: GapItem[] = [];

    for (const category of categories) {
      const schema = category.dbcFieldSchema as FieldSchema[];
      const extractedData = extracted.find(e => e.categoryCode === category.dbcCode);

      for (const field of schema) {
        if (field.required && !extractedData?.data?.[field.key]) {
          gaps.push({
            categoryCode: category.dbcCode,
            categoryName: category.dbcName,
            fieldKey: field.key,
            fieldLabel: field.label,
            severity: field.required ? 'CRITICAL' : 'INFO',
          });
        }
      }
    }

    return { totalGaps: gaps.length, criticalGaps: gaps.filter(g => g.severity === 'CRITICAL').length, items: gaps };
  }
}
```

```typescript
// 문서 생성 시 CONFIDENTIAL 데이터를 [REDACTED]로 마스킹 후 AI에 전달
// 생성 결과에서 [REDACTED]를 원본 데이터로 복원
private maskConfidentialData(data: Record<string, any>, categories: DocBaseCategoryEntity[]): MaskedData {
  const masked = { ...data };
  const restoreMap = new Map();

  for (const cat of categories) {
    if (cat.dbcConfidentiality === 'CONFIDENTIAL') {
      for (const [key, value] of Object.entries(masked[cat.dbcCode] || {})) {
        const placeholder = `[CONFIDENTIAL_${cat.dbcCode}_${key}]`;
        restoreMap.set(placeholder, value);
        masked[cat.dbcCode][key] = placeholder;
      }
    }
  }

  return { masked, restoreMap };
}
```
