// today.service.ts
async getMyToday(userId: string, entityId: string): Promise<MyTodayResponse> {
  const today = new Date().toISOString().split('T')[0];

  // 1) 오늘 마감 할일
  const todayTodos = await this.todoRepo.find({
    where: { usrId: userId, entId: entityId, tdoDueDate: today, tdoDeletedAt: IsNull() },
    relations: ['user'],
    order: { tdoCreatedAt: 'DESC' },
  });

  // 2) 지연 할일 (마감일 < 오늘 & 미완료)
  const overdueTodos = await this.todoRepo.find({
    where: {
      usrId: userId, entId: entityId,
      tdoDueDate: LessThan(today),
      tdoStatus: Not('COMPLETED'),
      tdoDeletedAt: IsNull(),
    },
    relations: ['user'],
    order: { tdoDueDate: 'ASC' },
  });

  // 3) 진행중 할일
  const inProgressTodos = await this.todoRepo.find({
    where: {
      usrId: userId, entId: entityId,
      tdoStatus: 'IN_PROGRESS',
      tdoDeletedAt: IsNull(),
    },
    relations: ['user'],
    order: { tdoDueDate: 'ASC' },
  });

  // 4) 미종료 이슈 카운트
  const issueCount = await this.issueRepo.count({
    where: [
      { issReporterId: userId, entId: entityId, issStatus: Not(In(['RESOLVED', 'CLOSED'])), issDeletedAt: IsNull() },
      { issAssigneeId: userId, entId: entityId, issStatus: Not(In(['RESOLVED', 'CLOSED'])), issDeletedAt: IsNull() },
    ],
  });

  return {
    todayDue: todayTodos,
    overdue: overdueTodos,
    inProgress: inProgressTodos,
    issueCount,
    summary: {
      todayDueCount: todayTodos.length,
      overdueCount: overdueTodos.length,
      inProgressCount: inProgressTodos.length,
      completedTodayCount: /* 오늘 완료한 할일 수 */,
    },
  };
}
```

// today.service.ts
async getAllToday(userId: string, entityId: string): Promise<AllTodayResponse> {
  // 1) EntityUserRole 기반 법인 전 직원 목록
  const members = await this.eurRepo.find({
    where: { entId: entityId, eurStatus: 'ACTIVE' },
    relations: ['user'],
  });

  // 2) 직원별 할일 집계 (배치 쿼리)
  const userIds = members.map(m => m.usrId);
  const todoCounts = await this.todoRepo
    .createQueryBuilder('t')
    .select('t.usr_id', 'userId')
    .addSelect(`SUM(CASE WHEN t.tdo_due_date = CURRENT_DATE THEN 1 ELSE 0 END)`, 'todayDue')
    .addSelect(`SUM(CASE WHEN t.tdo_due_date < CURRENT_DATE AND t.tdo_status != 'COMPLETED' THEN 1 ELSE 0 END)`, 'overdue')
    .addSelect(`SUM(CASE WHEN t.tdo_status = 'IN_PROGRESS' THEN 1 ELSE 0 END)`, 'inProgress')
    .where('t.usr_id IN (:...userIds)', { userIds })
    .andWhere('t.ent_id = :entityId', { entityId })
    .andWhere('t.tdo_deleted_at IS NULL')
    .groupBy('t.usr_id')
    .getRawMany();

  // 3) 직원별 이슈 카운트 (배치)
  // 4) Employee 정보 조인 (부서, 직급)
  // 5) Unit 정보 (unt_name, unt_level)

  return { members: [...], summary: { totalMembers, ... } };
}
```

**권한 체크:** `@Auth()` + 서비스 내에서 `isMaster(userId, entityId)` 또는 `isAdmin(user.level)` 확인
